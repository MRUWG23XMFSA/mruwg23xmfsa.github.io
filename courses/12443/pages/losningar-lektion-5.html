 //Beräkna fakulteten av ett heltal
 function fakultet(n){
    fak = 1;
    for (i=2; i<=n; i++){
         fak = fak * i;
    }
    return fak;

Kontrollera om en sträng är ett palindrom. För att jämföra två strängar bör vi ta bort alla mellanslag och göra om alla bokstäver till antingen versaler (stora bokstäver) eller gemener (små bokstäver)

 //Kolla om en sträng är ett palindrom, använder en funktion vi 
// skrivit tidigare (se presentation från lektion)

   function bakOFram(inString){
      back ="";
        for (var i = inString.length ; i>=0; i--){
             back = back + inString.charAt(i);
        } 
        return back;  
   }

   function noSpaces(inString){
   // Ta bort mellanslag från en sträng
       outString = "";
       for (var i = 0; i<= inString.length; i++){
           if (inString.charAt(i) != " "){
                outString += inString.charAt(i);
           }
       }
       return outString;
    }


    let str = "Ola Salo";
    // För att se om strängarna är lite måste vi göra om stränge
    // Vi tar först bort alla mellanslag i strängen
    
    let whitespaceRemoved = noSpaces(str);
    alert(whitespaceRemoved);
    //Sedan gör vi om allt till versaler
    let myText = whitespaceRemoved.toUpperCase();
    alert(myText);
    if (myText == bakOFram(myText)){
        alert(str + " är ett palindrom");
    }
