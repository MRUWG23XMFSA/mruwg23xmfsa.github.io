/* Skriv ut användarens initialer, i en enklare variant kan man fråga efter förnamn
och efternamn i varsin prompt. I den här lösningen måste vi dela på namnet.
Programmet fungerar inte om användaren börjar sitt namn med ett mellanslag...
*/
  
let namn = prompt("Vad heter du (för och efternamn)?");
  // Dela på för och efternamn vi antar att det finns ett mellanslag där vi ska dela
  // Resultatet blir en lista där förnamnet ligger i position 0 och efternamnet i position 1
let splitted=namn.split(" ");
let fornamn = splitted[0];
let efternamn = splitted[1];
  //Initialerna tar vi som den första bokstaven ur förnamn och efternamn.  
let initialer = fornamn[0] + efternamn[0];
text(10,20,15,initialer,"black");
