Här är en möjlig lösning på pacman uppgiften.

<!DOCTYPE html>
<html>
    <head>
        <title>Pacman</title>
        <meta charset="UTF-8" >
    </head>
    <body>
       <h1> Pacman</h1>
    <canvas id="pacmania" width="600" height="600"></canvas>
    <script>
        let canvas = document.getElementById("pacmania");
        let context = canvas.getContext("2d");
        

        function draw_pacman(x,y, radius, mouth) {
          angle = 0.2 * Math.PI * mouth;
          context.strokeStyle ="black"
          context.fillStyle = "Gold";
          context.beginPath();
          context.arc(x, y, radius, angle, 2 * Math.PI - angle);
          context.lineTo(x, y);
          context.closePath();
          context.fill();
          context.stroke();
        }


        draw_pacman(100, 200, 30, 0.5);
        draw_pacman(200, 300, 50, 0);
        draw_pacman(50, 300, 45, 1);
        
    </script>
    </body>
</html>
