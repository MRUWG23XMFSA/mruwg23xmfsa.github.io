Tärningen

Rita en tärning med hjälp av en funktion, funktioner gås igenom på lektion 5.

function ritaTarning(x,y, varde){
  // Rita en tärning med övre vänstra hörnet i x,y och en fast storlek på 100 pixlar
                  
  // Rita först en svart rektangel sedan en vit strax innanför
  rectangle(x,y, 100, 100,"black");
  rectangle(x + 3, y + 3, 94, 94, "white");
  // Nu ska vi rita olika antal ögon beroende på tärningens värde
  // Ett öga är en cirkel med radie 5
  switch (varde){
    case 1:
      // Ett ska sitta mitt i tärningen
      circle(x+50, y + 50, 5 ,"red");
      break;
    case 2:
      // Två prickar i varsitt "hörn"
      circle(x+75, y + 25, 5 ,"red");
      circle(x+25, y + 75, 5 ,"red");
      break;
    case 3:
      // Då ska vi rita de prickar som ingår i ett och två
      circle(x+50, y + 50, 5 ,"red");
      circle(x+75, y + 25, 5 ,"red");
      circle(x+25, y + 75, 5 ,"red");
      break;
    case 4:
      // Rita prickar i de fyra hörnen
      circle(x+75, y + 25, 5 ,"red");
      circle(x+25, y + 75, 5 ,"red");
      circle(x+75, y + 75, 5 ,"red");
      circle(x+25, y + 25, 5 ,"red");
      break;
    case 5:
      // Rita de fyra prickarna för fyra samt en i mitten
      circle(x+75, y + 25, 5 ,"red");
      circle(x+25, y + 75, 5 ,"red");
      circle(x+75, y + 75, 5 ,"red");
      circle(x+25, y + 25, 5 ,"red");
      circle(x+50, y + 50, 5 ,"red");
      break;
    case 6:
      // Rita två rader med tre prickar
      circle(x+75, y + 25, 5 ,"red");
      circle(x+25, y + 75, 5 ,"red");
      circle(x+75, y + 75, 5 ,"red");
      circle(x+25, y + 25, 5 ,"red");
      circle(x+50, y + 25, 5 ,"red");
      circle(x+50, y + 75, 5 ,"red");
  } // end case
}
                
function start(){
  ritaTarning(20,20,6);
}
